2.4 OLS Residuals Add up to Zero by Construction

# Brand new set of variables 
set.seed(1234)
N <- 10
x <- 9*rnorm(N)
u <- 36*rnorm(N)
y = 3 + 2*x + u 

m2 <- lm(y~x)
