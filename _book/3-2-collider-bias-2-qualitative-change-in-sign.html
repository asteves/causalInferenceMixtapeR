3.2 Collider Bias 2 Qualitative Change in Sign

library(estimatr) 
set.seed(541)

N <- 2500
z <- rnorm(N)
k <- rnorm(N, 10, 4)
d <- ifelse(k >=12, 1, 0)

# Treatment effect is 50. Notice that y is not a function of X 
y <- 50*d + 100 + rnorm(N)
x <- 50*d + y + rnorm(N, 50, 1)

collider2 <- data.frame(y =y, 
                        x= x,
                        d = d)

lm_robust(y~d, data = collider2, se_type = "stata")
